        const http = require('http');

        // Save original data for restoration
        const fs = require('fs');
        const original = fs.readFileSync('resume.json', 'utf-8');

        // Step 1: Verify GET /api/resume
        http.get('http://localhost:3000/api/resume', (res) => {
          let d = '';
          res.on('data', c => d += c);
          res.on('end', () => {
            const r = JSON.parse(d);
            const requiredFields = ['name', 'contact', 'summary', 'experience', 'projects', 'skills', 'education'];
            const missing = requiredFields.filter(f => !(f in r));
            if (missing.length > 0) throw new Error('Missing fields: ' + missing.join(', '));
            if (!Array.isArray(r.experience)) throw new Error('experience is not an array');
            if (!Array.isArray(r.skills)) throw new Error('skills is not an array');
            console.log('GET /api/resume OK: name=' + r.name + ', experience=' + r.experience.length + ' entries, skills=' + r.skills.length);
          });
        });

        // Step 2: Verify PUT /api/resume
        const testData = JSON.parse(original);
        testData._test_marker = 'verification_test';
        const putData = JSON.stringify(testData);
        const putReq = http.request({ hostname: 'localhost', port: 3000, path: '/api/resume', method: 'PUT', headers: { 'Content-Type': 'application/json', 'Content-Length': Buffer.byteLength(putData) } }, (res) => {
          let d = '';
          res.on('data', c => d += c);
          res.on('end', () => {
            const r = JSON.parse(d);
            if (!r.ok) throw new Error('PUT did not return ok:true');
            console.log('PUT /api/resume OK');

            // Step 3: Restore original data
            const restoreReq = http.request({ hostname: 'localhost', port: 3000, path: '/api/resume', method: 'PUT', headers: { 'Content-Type': 'application/json', 'Content-Length': Buffer.byteLength(original) } }, () => {});
            restoreReq.write(original);
            restoreReq.end();
            console.log('Original data restored');
          });
        });
        putReq.write(putData);
        putReq.end();
